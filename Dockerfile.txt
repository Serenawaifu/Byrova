# Use Python 3.8-slim base image
FROM python:3.8-slim

# Install necessary dependencies for Selenium and Chrome
RUN apt-get update && apt-get install -y wget gnupg2 curl unzip \
    && wget https://dl.google.com/linux/direct/google-chrome-stable_current_amd64.deb \
        && dpkg -i google-chrome-stable_current_amd64.deb \
            && apt-get -y -f install

            # Install ChromeDriver compatible with Chrome 131.0.6778.264
            RUN CHROME_VERSION=$(google-chrome --version | sed 's/[^0-9]*[0-9]*\..*/\1/') && \
                wget -q "https://chromedriver.storage.googleapis.com/131.0.6778.264/chromedriver_linux64.zip" -O chromedriver.zip && \
                    unzip chromedriver.zip && \
                        mv chromedriver /usr/local/bin && \
                            chmod +x /usr/local/bin/chromedriver

                            # Set the working directory inside the container
                            WORKDIR /app

                            # Copy the requirements.txt file and install Python dependencies
                            COPY requirements.txt requirements.txt
                            RUN pip install -r requirements.txt

                            # Copy the rest of the bot files into the container
                            COPY . .

                            # Set the command to run your bot
                            CMD ["python", "twitter_bot.py"]
